### 2. 折叠逻辑实现

```javascript
// 折叠状态管理（默认全部展开）
const expandedCategories = ref([
  'tools', 'ai-agent', 'erp', 'office', 
  'pm', 'hr', 'learning', 'dev'
])

// 切换折叠
const toggleCategory = (categoryId) => {
  const index = expandedCategories.value.indexOf(categoryId)
  if (index > -1) {
    expandedCategories.value.splice(index, 1)  // 折叠
  } else {
    expandedCategories.value.push(categoryId)  // 展开
  }
}

// 根据分类筛选平台
const getPlatformsByCategory = (categoryId) => {
  return workflowPlatforms.value.filter(p => p.category === categoryId)
}
```

### 3. 模板结构

```vue
<!-- 分类循环 -->
<div v-for="category in platformCategories" :key="category.id" 
     class="category-section">
  <!-- 分类头部（可点击折叠） -->
  <div class="category-header" @click="toggleCategory(category.id)">
    <div class="category-title">
      <span class="category-icon">{{ category.icon }}</span>
      <h3>{{ category.name }}</h3>
      <el-tag size="small">{{ category.count }}个</el-tag>
    </div>
    <el-icon class="toggle-icon" 
             :class="{ 'is-expanded': expandedCategories.includes(category.id) }">
      <ArrowDown />
    </el-icon>
  </div>
  
  <!-- 折叠内容区 -->
  <el-collapse-transition>
    <div v-show="expandedCategories.includes(category.id)" 
         class="category-content">
      <div class="workflow-platforms-grid">
        <!-- 平台卡片 -->
        <div v-for="platform in getPlatformsByCategory(category.id)" 
             :key="platform.id" class="workflow-platform-card">
          ...平台内容...
        </div>
      </div>
    </div>
  </el-collapse-transition>
</div>
```
