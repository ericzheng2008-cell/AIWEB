// 产品服务数据结构
const productSeries = computed(() => {

      path: '/tool-selector'
    },
    {
      id: 'fault-tracking',
      name: '故障工单管理',
      description: '全流程工单管理系统，从创建到追踪到完成',
      image: '...',
      type: 'service',
      path: '/fault-tracking'
    }
  ]
  
  return [...categories, ...additionalServices]
})

// 点击事件处理
const goToProducts = (series) => {
  if (series.type === 'service' && series.path) {
    router.push(series.path)  // 服务类直接跳转
  } else {
    router.push('/products-services')  // 产品分类跳转列表页
  }
}
