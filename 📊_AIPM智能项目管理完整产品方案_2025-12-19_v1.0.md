#### 项目健康指数（PHI）算法
```javascript
function calculatePHI(project) {
  const metrics = {
    进度健康度: calculateScheduleHealth(project),
    成本健康度: calculateCostHealth(project),
    质量健康度: calculateQualityHealth(project),
    风险健康度: calculateRiskHealth(project),
    团队健康度: calculateTeamHealth(project)
  }
  
  const weights = { 进度: 0.3, 成本: 0.25, 质量: 0.2, 风险: 0.15, 团队: 0.1 }
  
  const PHI = Object.entries(metrics).reduce((sum, [key, value]) => {
    return sum + value * weights[key]
  }, 0)
  
  return {
    PHI: Math.round(PHI),
    等级: PHI > 80 ? '优秀' : PHI > 60 ? '良好' : PHI > 40 ? '中等' : '警告',
    metrics
  }
}
```

### 3. 风险智能识别与闭环

#### 风险识别算法
```javascript
// 多维度风险识别引擎
class RiskDetectionEngine {
  // 1. 基于规则的风险识别
  ruleBasedDetection(project) {
    const risks = []
    
    // 进度风险
    if (project.进度偏差 > 10%) {
      risks.push({
        type: '进度风险',
        severity: '高',
        description: `进度偏差${project.进度偏差}%，超过阈值`
      })
    }
    
    // 资源风险
    if (project.资源占用率 > 90%) {
      risks.push({
        type: '资源风险',
        severity: '中',
        description: '资源占用率过高，可能影响其他项目'
      })
    }
    
    return risks
  }
  
  // 2. 基于AI的风险预测
  async aiBasedPrediction(project) {
    const features = extractFeatures(project)
    const prediction = await this.riskModel.predict(features)
    
    return {
      风险概率: prediction.probability,
      预计影响: prediction.impact,
      建议措施: prediction.recommendations
    }
  }
  
  // 3. 基于知识图谱的因果推理
  causalInference(project) {
    const risks = []
    
    // 查询知识图谱
    const relatedRisks = this.knowledgeGraph.query(`
      MATCH (p:Project {id: '${project.id}'})-[:HAS_RISK]->(r:Risk)
      WHERE r.probability > 0.7
      RETURN r
    `)
    
    return relatedRisks
  }
}
```
